168#ifndef EVENT__DISABLE_DEBUG_MODE
169/* These functions implement a hashtable of which 'struct event *' structures
170 * have been setup or added. We don't want to trust the content of the struct
171 * event itself, since we're trying to work through cases where an event gets
172 * clobbered or freed. Instead, we keep a hashtable indexed by the pointer.
173 */
174
175struct event_debug_entry {
176 HT_ENTRY(event_debug_entry) node;
177 const struct event *ptr;
178 unsigned added : 1;
179};
180
181static inline unsigned
182hash_debug_entry(const struct event_debug_entry *e)
183{
184 /* We need to do this silliness to convince compilers that we
185 * honestly mean to cast e->ptr to an integer, and discard any
186 * part of it that doesn't fit in an unsigned.
187 */
188 unsigned u = (unsigned) ((ev_uintptr_t) e->ptr);
189 /* Our hashtable implementation is pretty sensitive to low bits,
190 * and every struct event is over 64 bytes in size, so we can
191 * just say >>6. */
192 return (u >> 6);
193}
194
195static inline int
196eq_debug_entry(const struct event_debug_entry *a,
197 const struct event_debug_entry *b)
198{
199 return a->ptr == b->ptr;
200}

224static HT_HEAD(event_debug_map, event_debug_entry) global_debug_map =

227HT_PROTOTYPE(event_debug_map, event_debug_entry, node, hash_debug_entry,

232/* Macro: record that ev is now setup (that is, ready for an add) */
233#define event_debug_note_setup_(ev) do { \
234 if (event_debug_mode_on_) { \
235 struct event_debug_entry *dent,find; \
236 find.ptr = (ev); \
237 EVLOCK_LOCK(event_debug_map_lock_, 0); \
238 dent = HT_FIND(event_debug_map, &global_debug_map, &find); \
239 if (dent) { \
240 dent->added = 0; \
241 } else { \
242 dent = mm_malloc(sizeof(*dent)); \
243 if (!dent) \
244 event_err(1, \
245 "Out of memory in debugging code"); \
246 dent->ptr = (ev); \
247 dent->added = 0; \
248 HT_INSERT(event_debug_map, &global_debug_map, dent); \
249 } \
250 EVLOCK_UNLOCK(event_debug_map_lock_, 0); \
251 } \
252 event_debug_mode_too_late = 1; \
253 } while (0)
254/* Macro: record that ev is no longer setup */
255#define event_debug_note_teardown_(ev) do { \
256 if (event_debug_mode_on_) { \
257 struct event_debug_entry *dent,find; \
258 find.ptr = (ev); \
259 EVLOCK_LOCK(event_debug_map_lock_, 0); \
260 dent = HT_REMOVE(event_debug_map, &global_debug_map, &find); \
261 if (dent) \
262 mm_free(dent); \
263 EVLOCK_UNLOCK(event_debug_map_lock_, 0); \
264 } \
265 event_debug_mode_too_late = 1; \
266 } while (0)
267/* Macro: record that ev is now added */
268#define event_debug_note_add_(ev) do { \
269 if (event_debug_mode_on_) { \
270 struct event_debug_entry *dent,find; \
271 find.ptr = (ev); \
272 EVLOCK_LOCK(event_debug_map_lock_, 0); \
273 dent = HT_FIND(event_debug_map, &global_debug_map, &find); \
274 if (dent) { \
275 dent->added = 1; \
276 } else { \
277 event_errx(EVENT_ERR_ABORT_, \
278 "%s: noting an add on a non-setup event %p" \
279 " (events: 0x%x, fd: "EV_SOCK_FMT \
280 ", flags: 0x%x)", \
281 __func__, (ev), (ev)->ev_events, \
282 EV_SOCK_ARG((ev)->ev_fd), (ev)->ev_flags); \
283 } \
284 EVLOCK_UNLOCK(event_debug_map_lock_, 0); \
285 } \
286 event_debug_mode_too_late = 1; \
287 } while (0)
288/* Macro: record that ev is no longer added */
289#define event_debug_note_del_(ev) do { \
290 if (event_debug_mode_on_) { \
291 struct event_debug_entry *dent,find; \
292 find.ptr = (ev); \
293 EVLOCK_LOCK(event_debug_map_lock_, 0); \
294 dent = HT_FIND(event_debug_map, &global_debug_map, &find); \
295 if (dent) { \
296 dent->added = 0; \
297 } else { \
298 event_errx(EVENT_ERR_ABORT_, \
299 "%s: noting a del on a non-setup event %p" \
300 " (events: 0x%x, fd: "EV_SOCK_FMT \
301 ", flags: 0x%x)", \
302 __func__, (ev), (ev)->ev_events, \
303 EV_SOCK_ARG((ev)->ev_fd), (ev)->ev_flags); \
304 } \
305 EVLOCK_UNLOCK(event_debug_map_lock_, 0); \
306 } \
307 event_debug_mode_too_late = 1; \
308 } while (0)
309/* Macro: assert that ev is setup (i.e., okay to add or inspect) */
310#define event_debug_assert_is_setup_(ev) do { \
311 if (event_debug_mode_on_) { \
312 struct event_debug_entry *dent,find; \
313 find.ptr = (ev); \
314 EVLOCK_LOCK(event_debug_map_lock_, 0); \
315 dent = HT_FIND(event_debug_map, &global_debug_map, &find); \
316 if (!dent) { \
317 event_errx(EVENT_ERR_ABORT_, \
318 "%s called on a non-initialized event %p" \
319 " (events: 0x%x, fd: "EV_SOCK_FMT\
320 ", flags: 0x%x)", \
321 __func__, (ev), (ev)->ev_events, \
322 EV_SOCK_ARG((ev)->ev_fd), (ev)->ev_flags); \
323 } \
324 EVLOCK_UNLOCK(event_debug_map_lock_, 0); \
325 } \
326 } while (0)
327/* Macro: assert that ev is not added (i.e., okay to tear down or set
328 * up again) */
329#define event_debug_assert_not_added_(ev) do { \
330 if (event_debug_mode_on_) { \
331 struct event_debug_entry *dent,find; \
332 find.ptr = (ev); \
333 EVLOCK_LOCK(event_debug_map_lock_, 0); \
334 dent = HT_FIND(event_debug_map, &global_debug_map, &find); \
335 if (dent && dent->added) { \
336 event_errx(EVENT_ERR_ABORT_, \
337 "%s called on an already added event %p" \
338 " (events: 0x%x, fd: "EV_SOCK_FMT", " \
339 "flags: 0x%x)", \
340 __func__, (ev), (ev)->ev_events, \
341 EV_SOCK_ARG((ev)->ev_fd), (ev)->ev_flags); \
342 } \
343 EVLOCK_UNLOCK(event_debug_map_lock_, 0); \
344 } \
345 } while (0)
346#else
347#define event_debug_note_setup_(ev) \
348 ((void)0)

364/* How often (in seconds) do we check for changes in wall clock time relative
365 * to monotonic time? Set this to -1 for 'never.' */
366#define CLOCK_SYNC_INTERVAL 5
367
368/** Set 'tp' to the current time according to 'base'. We must hold the lock
369 * on 'base'. If there is a cached time, return it. Otherwise, use
370 * clock_gettime or gettimeofday as appropriate to find out the right time.
371 * Return 0 on success, -1 on failure.
372 */
373static int
374gettime(struct event_base *base, struct timeval *tp)
375{
376 EVENT_BASE_ASSERT_LOCKED(base);
377
378 if (base->tv_cache.tv_sec) {
379 *tp = base->tv_cache;
380 return (0);
381 }
382
383 if (evutil_gettime_monotonic_(&base->monotonic_timer, tp) == -1) {
384 return -1;
385 }
386
387 if (base->last_updated_clock_diff + CLOCK_SYNC_INTERVAL
388 < tp->tv_sec) {
389 struct timeval tv;
390 evutil_gettimeofday(&tv,NULL);
391 evutil_timersub(&tv, tp, &base->tv_clock_diff);
392 base->last_updated_clock_diff = tp->tv_sec;
393 }
394
395 return 0;
396}

398int
399event_base_gettimeofday_cached(struct event_base *base, struct timeval *tv)
400{
401 int r;
402 if (!base) {
403 base = current_base;
404 if (!current_base)
405 return evutil_gettimeofday(tv, NULL);
406 }
407
408 EVBASE_ACQUIRE_LOCK(base, th_base_lock);
409 if (base->tv_cache.tv_sec == 0) {
410 r = evutil_gettimeofday(tv, NULL);
411 } else {
412 evutil_timeradd(&base->tv_cache, &base->tv_clock_diff, tv);
413 r = 0;
414 }
415 EVBASE_RELEASE_LOCK(base, th_base_lock);
416 return r;
417}

492/** Return true iff 'method' is the name of a method that 'cfg' tells us to
493 * avoid. */
494static int
495event_config_is_avoided_method(const struct event_config *cfg,
496 const char *method)
497{
498 struct event_config_entry *entry;
499
500 TAILQ_FOREACH(entry, &cfg->entries, next) {
501 if (entry->avoid_method != NULL &&
502 strcmp(entry->avoid_method, method) == 0)
503 return (1);
504 }
505
506 return (0);
507}

509/** Return true iff 'method' is disabled according to the environment. */
510static int
511event_is_method_disabled(const char *name)
512{
513 char environment[64];
514 int i;
515
516 evutil_snprintf(environment, sizeof(environment), "EVENT_NO%s", name);
517 for (i = 8; environment[i] != '\0'; ++i)
518 environment[i] = EVUTIL_TOUPPER_(environment[i]);
519 /* Note that evutil_getenv_() ignores the environment entirely if
520 * we're setuid */
521 return (evutil_getenv_(environment) != NULL);
522}

565struct event_base *
566event_base_new_with_config(const struct event_config *cfg)
567{
568 int i;
569 struct event_base *base;
570 int should_check_environment;
571
572#ifndef EVENT__DISABLE_DEBUG_MODE
573 event_debug_mode_too_late = 1;
574#endif
575
576 if ((base = mm_calloc(1, sizeof(struct event_base))) == NULL) {
577 event_warn("%s: calloc", __func__);
578 return NULL;
579 }
580
581 if (cfg)
582 base->flags = cfg->flags;
583
584 should_check_environment =
585 !(cfg && (cfg->flags & EVENT_BASE_FLAG_IGNORE_ENV));
586
587 {
588 struct timeval tmp;
589 int precise_time =
590 cfg && (cfg->flags & EVENT_BASE_FLAG_PRECISE_TIMER);
591 int flags;
592 if (should_check_environment && !precise_time) {
593 precise_time = evutil_getenv_("EVENT_PRECISE_TIMER") != NULL;
594 base->flags |= EVENT_BASE_FLAG_PRECISE_TIMER;
595 }
596 flags = precise_time ? EV_MONOT_PRECISE : 0;
597 evutil_configure_monotonic_time_(&base->monotonic_timer, flags);
598
599 gettime(base, &tmp);
600 }
601
602 min_heap_ctor_(&base->timeheap);
603
604 base->sig.ev_signal_pair[0] = -1;
605 base->sig.ev_signal_pair[1] = -1;
606 base->th_notify_fd[0] = -1;
607 base->th_notify_fd[1] = -1;
608
609 TAILQ_INIT(&base->active_later_queue);
610
611 evmap_io_initmap_(&base->io);
612 evmap_signal_initmap_(&base->sigmap);
613 event_changelist_init_(&base->changelist);
614
615 base->evbase = NULL;
616
617 if (cfg) {
618 memcpy(&base->max_dispatch_time,
619 &cfg->max_dispatch_interval, sizeof(struct timeval));
620 base->limit_callbacks_after_prio =
621 cfg->limit_callbacks_after_prio;
622 } else {
623 base->max_dispatch_time.tv_sec = -1;
624 base->limit_callbacks_after_prio = 1;
625 }
626 if (cfg && cfg->max_dispatch_callbacks >= 0) {
627 base->max_dispatch_callbacks = cfg->max_dispatch_callbacks;
628 } else {
629 base->max_dispatch_callbacks = INT_MAX;
630 }
631 if (base->max_dispatch_callbacks == INT_MAX &&
632 base->max_dispatch_time.tv_sec == -1)
633 base->limit_callbacks_after_prio = INT_MAX;
634
635 for (i = 0; eventops[i] && !base->evbase; i++) {
636 if (cfg != NULL) {
637 /* determine if this backend should be avoided */
638 if (event_config_is_avoided_method(cfg,
639 eventops[i]->name))
640 continue;
641 if ((eventops[i]->features & cfg->require_features)
642 != cfg->require_features)
643 continue;
644 }
645
646 /* also obey the environment variables */
647 if (should_check_environment &&
648 event_is_method_disabled(eventops[i]->name))
649 continue;
650
651 base->evsel = eventops[i];
652
653 base->evbase = base->evsel->init(base);
654 }
655
656 if (base->evbase == NULL) {
657 event_warnx("%s: no event mechanism available",
658 __func__);
659 base->evsel = NULL;
660 event_base_free(base);
661 return NULL;
662 }
663
664 if (evutil_getenv_("EVENT_SHOW_METHOD"))
665 event_msgx("libevent using: %s", base->evsel->name);
666
667 /* allocate a single active event queue */
668 if (event_base_priority_init(base, 1) < 0) {
669 event_base_free(base);
670 return NULL;
671 }
672
673 /* prepare for threading */
674
675#if !defined(EVENT__DISABLE_THREAD_SUPPORT) && !defined(EVENT__DISABLE_DEBUG_MODE)
676 event_debug_created_threadable_ctx_ = 1;
677#endif
678
679#ifndef EVENT__DISABLE_THREAD_SUPPORT
680 if (EVTHREAD_LOCKING_ENABLED() &&
681 (!cfg || !(cfg->flags & EVENT_BASE_FLAG_NOLOCK))) {
682 int r;
683 EVTHREAD_ALLOC_LOCK(base->th_base_lock, 0);
684 EVTHREAD_ALLOC_COND(base->current_event_cond);
685 r = evthread_make_base_notifiable(base);
686 if (r<0) {
687 event_warnx("%s: Unable to make base notifiable.", __func__);
688 event_base_free(base);
689 return NULL;
690 }
691 }
692#endif
693
694#ifdef _WIN32
695 if (cfg && (cfg->flags & EVENT_BASE_FLAG_STARTUP_IOCP))
696 event_base_start_iocp_(base, cfg->n_cpus_hint);

733static int
734event_base_cancel_single_callback_(struct event_base *base,
735 struct event_callback *evcb,
736 int run_finalizers)
737{
738 int result = 0;
739
740 if (evcb->evcb_flags & EVLIST_INIT) {
741 struct event *ev = event_callback_to_event(evcb);
742 if (!(ev->ev_flags & EVLIST_INTERNAL)) {
743 event_del_(ev, EVENT_DEL_EVEN_IF_FINALIZING);
744 result = 1;
745 }
746 } else {
747 EVBASE_ACQUIRE_LOCK(base, th_base_lock);
748 event_callback_cancel_nolock_(base, evcb, 1);
749 EVBASE_RELEASE_LOCK(base, th_base_lock);
750 result = 1;
751 }
752
753 if (run_finalizers && (evcb->evcb_flags & EVLIST_FINALIZING)) {
754 switch (evcb->evcb_closure) {
755 case EV_CLOSURE_EVENT_FINALIZE:
756 case EV_CLOSURE_EVENT_FINALIZE_FREE: {
757 struct event *ev = event_callback_to_event(evcb);
758 ev->ev_evcallback.evcb_cb_union.evcb_evfinalize(ev, ev->ev_arg);
759 if (evcb->evcb_closure == EV_CLOSURE_EVENT_FINALIZE_FREE)
760 mm_free(ev);
761 break;
762 }
763 case EV_CLOSURE_CB_FINALIZE:
764 evcb->evcb_cb_union.evcb_cbfinalize(evcb, evcb->evcb_arg);
765 break;
766 default:
767 break;
768 }
769 }
770 return result;
771}

773static int event_base_free_queues_(struct event_base *base, int run_finalizers)
774{
775 int deleted = 0, i;
776
777 for (i = 0; i < base->nactivequeues; ++i) {
778 struct event_callback *evcb, *next;
779 for (evcb = TAILQ_FIRST(&base->activequeues[i]); evcb; ) {
780 next = TAILQ_NEXT(evcb, evcb_active_next);
781 deleted += event_base_cancel_single_callback_(base, evcb, run_finalizers);
782 evcb = next;
783 }
784 }
785
786 {
787 struct event_callback *evcb;
788 while ((evcb = TAILQ_FIRST(&base->active_later_queue))) {
789 deleted += event_base_cancel_single_callback_(base, evcb, run_finalizers);
790 }
791 }
792
793 return deleted;
794}

796static void
797event_base_free_(struct event_base *base, int run_finalizers)
798{
799 int i, n_deleted=0;
800 struct event *ev;
801 /* XXXX grab the lock? If there is contention when one thread frees
802 * the base, then the contending thread will be very sad soon. */
803
804 /* event_base_free(NULL) is how to free the current_base if we
805 * made it with event_init and forgot to hold a reference to it. */
806 if (base == NULL && current_base)
807 base = current_base;
808 /* Don't actually free NULL. */
809 if (base == NULL) {
810 event_warnx("%s: no base to free", __func__);
811 return;
812 }
813 /* XXX(niels) - check for internal events first */
814
815#ifdef _WIN32
816 event_base_stop_iocp_(base);
817#endif
818
819 /* threading fds if we have them */
820 if (base->th_notify_fd[0] != -1) {
821 event_del(&base->th_notify);
822 EVUTIL_CLOSESOCKET(base->th_notify_fd[0]);
823 if (base->th_notify_fd[1] != -1)
824 EVUTIL_CLOSESOCKET(base->th_notify_fd[1]);
825 base->th_notify_fd[0] = -1;
826 base->th_notify_fd[1] = -1;
827 event_debug_unassign(&base->th_notify);
828 }
829
830 /* Delete all non-internal events. */
831 evmap_delete_all_(base);
832
833 while ((ev = min_heap_top_(&base->timeheap)) != NULL) {
834 event_del(ev);
835 ++n_deleted;
836 }
837 for (i = 0; i < base->n_common_timeouts; ++i) {
838 struct common_timeout_list *ctl =
839 base->common_timeout_queues[i];
840 event_del(&ctl->timeout_event); /* Internal; doesn't count */
841 event_debug_unassign(&ctl->timeout_event);
842 for (ev = TAILQ_FIRST(&ctl->events); ev; ) {
843 struct event *next = TAILQ_NEXT(ev,
844 ev_timeout_pos.ev_next_with_common_timeout);
845 if (!(ev->ev_flags & EVLIST_INTERNAL)) {
846 event_del(ev);
847 ++n_deleted;
848 }
849 ev = next;
850 }
851 mm_free(ctl);
852 }
853 if (base->common_timeout_queues)
854 mm_free(base->common_timeout_queues);
855
856 for (;;) {
857 /* For finalizers we can register yet another finalizer out from
858 * finalizer, and iff finalizer will be in active_later_queue we can
859 * add finalizer to activequeues, and we will have events in
860 * activequeues after this function returns, which is not what we want
861 * (we even have an assertion for this).
862 *
863 * A simple case is bufferevent with underlying (i.e. filters).
864 */
865 int i = event_base_free_queues_(base, run_finalizers);
866 if (!i) {
867 break;
868 }
869 n_deleted += i;
870 }
871
872 if (n_deleted)
873 event_debug(("%s: %d events were still set in base",
874 __func__, n_deleted));
875
876 while (LIST_FIRST(&base->once_events)) {
877 struct event_once *eonce = LIST_FIRST(&base->once_events);
878 LIST_REMOVE(eonce, next_once);
879 mm_free(eonce);
880 }
881
882 if (base->evsel != NULL && base->evsel->dealloc != NULL)
883 base->evsel->dealloc(base);
884
885 for (i = 0; i < base->nactivequeues; ++i)
886 EVUTIL_ASSERT(TAILQ_EMPTY(&base->activequeues[i]));
887
888 EVUTIL_ASSERT(min_heap_empty_(&base->timeheap));
889 min_heap_dtor_(&base->timeheap);
890
891 mm_free(base->activequeues);
892
893 evmap_io_clear_(&base->io);
894 evmap_signal_clear_(&base->sigmap);
895 event_changelist_freemem_(&base->changelist);
896
897 EVTHREAD_FREE_LOCK(base->th_base_lock, 0);
898 EVTHREAD_FREE_COND(base->current_event_cond);
899
900 /* If we're freeing current_base, there won't be a current_base. */
901 if (base == current_base)
902 current_base = NULL;
903 mm_free(base);
904}

937/* reinitialize the event base after a fork */
938int
939event_reinit(struct event_base *base)
940{
941 const struct eventop *evsel;
942 int res = 0;
943 int was_notifiable = 0;
944 int had_signal_added = 0;
945
946 EVBASE_ACQUIRE_LOCK(base, th_base_lock);
947
948 evsel = base->evsel;
949
950 /* check if this event mechanism requires reinit on the backend */
951 if (evsel->need_reinit) {
952 /* We're going to call event_del() on our notify events (the
953 * ones that tell about signals and wakeup events). But we
954 * don't actually want to tell the backend to change its
955 * state, since it might still share some resource (a kqueue,
956 * an epoll fd) with the parent process, and we don't want to
957 * delete the fds from _that_ backend, we temporarily stub out
958 * the evsel with a replacement.
959 */
960 base->evsel = &nil_eventop;
961 }
962
963 /* We need to re-create a new signal-notification fd and a new
964 * thread-notification fd. Otherwise, we'll still share those with
965 * the parent process, which would make any notification sent to them
966 * get received by one or both of the event loops, more or less at
967 * random.
968 */
969 if (base->sig.ev_signal_added) {
970 event_del_nolock_(&base->sig.ev_signal, EVENT_DEL_AUTOBLOCK);
971 event_debug_unassign(&base->sig.ev_signal);
972 memset(&base->sig.ev_signal, 0, sizeof(base->sig.ev_signal));
973 had_signal_added = 1;
974 base->sig.ev_signal_added = 0;
975 }
976 if (base->sig.ev_signal_pair[0] != -1)
977 EVUTIL_CLOSESOCKET(base->sig.ev_signal_pair[0]);
978 if (base->sig.ev_signal_pair[1] != -1)
979 EVUTIL_CLOSESOCKET(base->sig.ev_signal_pair[1]);
980 if (base->th_notify_fn != NULL) {
981 was_notifiable = 1;
982 base->th_notify_fn = NULL;
983 }
984 if (base->th_notify_fd[0] != -1) {
985 event_del_nolock_(&base->th_notify, EVENT_DEL_AUTOBLOCK);
986 EVUTIL_CLOSESOCKET(base->th_notify_fd[0]);
987 if (base->th_notify_fd[1] != -1)
988 EVUTIL_CLOSESOCKET(base->th_notify_fd[1]);
989 base->th_notify_fd[0] = -1;
990 base->th_notify_fd[1] = -1;
991 event_debug_unassign(&base->th_notify);
992 }
993
994 /* Replace the original evsel. */
995 base->evsel = evsel;
996
997 if (evsel->need_reinit) {
998 /* Reconstruct the backend through brute-force, so that we do
999 * not share any structures with the parent process. For some
1000 * backends, this is necessary: epoll and kqueue, for
1001 * instance, have events associated with a kernel
1002 * structure. If didn't reinitialize, we'd share that
1003 * structure with the parent process, and any changes made by
1004 * the parent would affect our backend's behavior (and vice
1005 * versa).
1006 */
1007 if (base->evsel->dealloc != NULL)
1008 base->evsel->dealloc(base);
1009 base->evbase = evsel->init(base);
1010 if (base->evbase == NULL) {
1011 event_errx(1,
1012 "%s: could not reinitialize event mechanism",
1013 __func__);
1014 res = -1;
1015 goto done;
1016 }
1017
1018 /* Empty out the changelist (if any): we are starting from a
1019 * blank slate. */
1020 event_changelist_freemem_(&base->changelist);
1021
1022 /* Tell the event maps to re-inform the backend about all
1023 * pending events. This will make the signal notification
1024 * event get re-created if necessary. */
1025 if (evmap_reinit_(base) < 0)
1026 res = -1;
1027 } else {
1028 res = evsig_init_(base);
1029 if (res == 0 && had_signal_added) {
1030 res = event_add_nolock_(&base->sig.ev_signal, NULL, 0);
1031 if (res == 0)
1032 base->sig.ev_signal_added = 1;
1033 }
1034 }
1035
1036 /* If we were notifiable before, and nothing just exploded, become
1037 * notifiable again. */
1038 if (was_notifiable && res == 0)
1039 res = evthread_make_base_notifiable_nolock_(base);
1040
1041done:
1042 EVBASE_RELEASE_LOCK(base, th_base_lock);
1043 return (res);
1044}

1311static inline void
1312event_signal_closure(struct event_base *base, struct event *ev)
1313{

1335 ev->ev_pncalls = NULL;

1341/* Common timeouts are special timeouts that are handled as queues rather than
1342 * in the minheap. This is more efficient than the minheap if we happen to
1343 * know that we're going to get several thousands of timeout events all with
1344 * the same timeout value.
1345 *
1346 * Since all our timeout handling code assumes timevals can be copied,
1347 * assigned, etc, we can't use "magic pointer" to encode these common
1348 * timeouts. Searching through a list to see if every timeout is common could
1349 * also get inefficient. Instead, we take advantage of the fact that tv_usec
1350 * is 32 bits long, but only uses 20 of those bits (since it can never be over
1351 * 999999.) We use the top bits to encode 4 bites of magic number, and 8 bits
1352 * of index into the event_base's aray of common timeouts.
1353 */
1354
1355#define MICROSECONDS_MASK COMMON_TIMEOUT_MICROSECONDS_MASK
1356#define COMMON_TIMEOUT_IDX_MASK 0x0ff00000
1357#define COMMON_TIMEOUT_IDX_SHIFT 20
1358#define COMMON_TIMEOUT_MASK 0xf0000000
1359#define COMMON_TIMEOUT_MAGIC 0x50000000
1360
1361#define COMMON_TIMEOUT_IDX(tv) \
1362 (((tv)->tv_usec & COMMON_TIMEOUT_IDX_MASK)>>COMMON_TIMEOUT_IDX_SHIFT)
1363
1364/** Return true iff if 'tv' is a common timeout in 'base' */
1365static inline int
1366is_common_timeout(const struct timeval *tv,
1367 const struct event_base *base)
1368{
1369 int idx;
1370 if ((tv->tv_usec & COMMON_TIMEOUT_MASK) != COMMON_TIMEOUT_MAGIC)
1371 return 0;
1372 idx = COMMON_TIMEOUT_IDX(tv);
1373 return idx < base->n_common_timeouts;
1374}
1375
1376/* True iff tv1 and tv2 have the same common-timeout index, or if neither
1377 * one is a common timeout. */
1378static inline int
1379is_same_common_timeout(const struct timeval *tv1, const struct timeval *tv2)
1380{
1381 return (tv1->tv_usec & ~MICROSECONDS_MASK) ==
1382 (tv2->tv_usec & ~MICROSECONDS_MASK);
1383}
1384
1385/** Requires that 'tv' is a common timeout. Return the corresponding
1386 * common_timeout_list. */
1387static inline struct common_timeout_list *
1388get_common_timeout_list(struct event_base *base, const struct timeval *tv)
1389{
1390 return base->common_timeout_queues[COMMON_TIMEOUT_IDX(tv)];
1391}

1405/* Add the timeout for the first event in given common timeout list to the
1406 * event_base's minheap. */
1407static void
1408common_timeout_schedule(struct common_timeout_list *ctl,
1409 const struct timeval *now, struct event *head)
1410{
1411 struct timeval timeout = head->ev_timeout;
1412 timeout.tv_usec &= MICROSECONDS_MASK;
1413 event_add_nolock_(&ctl->timeout_event, &timeout, 1);
1414}

1416/* Callback: invoked when the timeout for a common timeout queue triggers.
1417 * This means that (at least) the first event in that queue should be run,
1418 * and the timeout should be rescheduled if there are more events. */
1419static void
1420common_timeout_callback(evutil_socket_t fd, short what, void *arg)
1421{
1422 struct timeval now;
1423 struct common_timeout_list *ctl = arg;
1424 struct event_base *base = ctl->base;
1425 struct event *ev = NULL;
1426 EVBASE_ACQUIRE_LOCK(base, th_base_lock);
1427 gettime(base, &now);
1428 while (1) {
1429 ev = TAILQ_FIRST(&ctl->events);
1430 if (!ev || ev->ev_timeout.tv_sec > now.tv_sec ||
1431 (ev->ev_timeout.tv_sec == now.tv_sec &&
1432 (ev->ev_timeout.tv_usec&MICROSECONDS_MASK) > now.tv_usec))
1433 break;
1434 event_del_nolock_(ev, EVENT_DEL_NOBLOCK);
1435 event_active_nolock_(ev, EV_TIMEOUT, 1);
1436 }
1437 if (ev)
1438 common_timeout_schedule(ctl, &now, ev);
1439 EVBASE_RELEASE_LOCK(base, th_base_lock);
1440}

1442#define MAX_COMMON_TIMEOUTS 256
1443
1444const struct timeval *
1445event_base_init_common_timeout(struct event_base *base,
1446 const struct timeval *duration)
1447{
1448 int i;
1449 struct timeval tv;
1450 const struct timeval *result=NULL;
1451 struct common_timeout_list *new_ctl;
1452
1453 EVBASE_ACQUIRE_LOCK(base, th_base_lock);
1454 if (duration->tv_usec > 1000000) {
1455 memcpy(&tv, duration, sizeof(struct timeval));
1456 if (is_common_timeout(duration, base))
1457 tv.tv_usec &= MICROSECONDS_MASK;
1458 tv.tv_sec += tv.tv_usec / 1000000;
1459 tv.tv_usec %= 1000000;
1460 duration = &tv;
1461 }
1462 for (i = 0; i < base->n_common_timeouts; ++i) {
1463 const struct common_timeout_list *ctl =
1464 base->common_timeout_queues[i];
1465 if (duration->tv_sec == ctl->duration.tv_sec &&
1466 duration->tv_usec ==
1467 (ctl->duration.tv_usec & MICROSECONDS_MASK)) {
1468 EVUTIL_ASSERT(is_common_timeout(&ctl->duration, base));
1469 result = &ctl->duration;
1470 goto done;
1471 }
1472 }
1473 if (base->n_common_timeouts == MAX_COMMON_TIMEOUTS) {
1474 event_warnx("%s: Too many common timeouts already in use; "
1475 "we only support %d per event_base", __func__,
1476 MAX_COMMON_TIMEOUTS);
1477 goto done;
1478 }
1479 if (base->n_common_timeouts_allocated == base->n_common_timeouts) {
1480 int n = base->n_common_timeouts < 16 ? 16 :
1481 base->n_common_timeouts*2;
1482 struct common_timeout_list **newqueues =
1483 mm_realloc(base->common_timeout_queues,
1484 n*sizeof(struct common_timeout_queue *));
1485 if (!newqueues) {
1486 event_warn("%s: realloc",__func__);
1487 goto done;
1488 }
1489 base->n_common_timeouts_allocated = n;
1490 base->common_timeout_queues = newqueues;
1491 }
1492 new_ctl = mm_calloc(1, sizeof(struct common_timeout_list));
1493 if (!new_ctl) {
1494 event_warn("%s: calloc",__func__);
1495 goto done;
1496 }
1497 TAILQ_INIT(&new_ctl->events);
1498 new_ctl->duration.tv_sec = duration->tv_sec;
1499 new_ctl->duration.tv_usec =
1500 duration->tv_usec | COMMON_TIMEOUT_MAGIC |
1501 (base->n_common_timeouts << COMMON_TIMEOUT_IDX_SHIFT);
1502 evtimer_assign(&new_ctl->timeout_event, base,
1503 common_timeout_callback, new_ctl);
1504 new_ctl->timeout_event.ev_flags |= EVLIST_INTERNAL;
1505 event_priority_set(&new_ctl->timeout_event, 0);
1506 new_ctl->base = base;
1507 base->common_timeout_queues[base->n_common_timeouts++] = new_ctl;
1508 result = &new_ctl->duration;
1509
1510done:
1511 if (result)
1512 EVUTIL_ASSERT(is_common_timeout(result, base));
1513
1514 EVBASE_RELEASE_LOCK(base, th_base_lock);
1515 return result;
1516}

1518/* Closure function invoked when we're activating a persistent event. */
1519static inline void
1520event_persist_closure(struct event_base *base, struct event *ev)
1521{
1522 void (*evcb_callback)(evutil_socket_t, short, void *);
1523
1524 // Other fields of *ev that must be stored before executing
1525 evutil_socket_t evcb_fd;
1526 short evcb_res;
1527 void *evcb_arg;
1528
1529 /* reschedule the persistent event if we have a timeout. */
1530 if (ev->ev_io_timeout.tv_sec || ev->ev_io_timeout.tv_usec) {
1531 /* If there was a timeout, we want it to run at an interval of
1532 * ev_io_timeout after the last time it was _scheduled_ for,
1533 * not ev_io_timeout after _now_. If it fired for another
1534 * reason, though, the timeout ought to start ticking _now_. */
1535 struct timeval run_at, relative_to, delay, now;
1536 ev_uint32_t usec_mask = 0;
1537 EVUTIL_ASSERT(is_same_common_timeout(&ev->ev_timeout,
1538 &ev->ev_io_timeout));
1539 gettime(base, &now);
1540 if (is_common_timeout(&ev->ev_timeout, base)) {
1541 delay = ev->ev_io_timeout;
1542 usec_mask = delay.tv_usec & ~MICROSECONDS_MASK;
1543 delay.tv_usec &= MICROSECONDS_MASK;
1544 if (ev->ev_res & EV_TIMEOUT) {
1545 relative_to = ev->ev_timeout;
1546 relative_to.tv_usec &= MICROSECONDS_MASK;
1547 } else {
1548 relative_to = now;
1549 }
1550 } else {
1551 delay = ev->ev_io_timeout;
1552 if (ev->ev_res & EV_TIMEOUT) {
1553 relative_to = ev->ev_timeout;
1554 } else {
1555 relative_to = now;
1556 }
1557 }
1558 evutil_timeradd(&relative_to, &delay, &run_at);
1559 if (evutil_timercmp(&run_at, &now, <)) {
1560 /* Looks like we missed at least one invocation due to
1561 * a clock jump, not running the event loop for a
1562 * while, really slow callbacks, or
1563 * something. Reschedule relative to now.
1564 */
1565 evutil_timeradd(&now, &delay, &run_at);
1566 }
1567 run_at.tv_usec |= usec_mask;
1568 event_add_nolock_(ev, &run_at, 1);
1569 }
1570
1571 // Save our callback before we release the lock
1572 evcb_callback = ev->ev_callback;
1573 evcb_fd = ev->ev_fd;
1574 evcb_res = ev->ev_res;
1575 evcb_arg = ev->ev_arg;
1576
1577 // Release the lock
1578 EVBASE_RELEASE_LOCK(base, th_base_lock);
1579
1580 // Execute the callback
1581 (evcb_callback)(evcb_fd, evcb_res, evcb_arg);
1582}

1584/*
1585 Helper for event_process_active to process all the events in a single queue,
1586 releasing the lock as we go. This function requires that the lock be held
1587 when it's invoked. Returns -1 if we get a signal or an event_break that
1588 means we should stop processing any active events now. Otherwise returns
1589 the number of non-internal event_callbacks that we processed.
1590*/
1591static int
1592event_process_active_single_queue(struct event_base *base,
1593 struct evcallback_list *activeq,
1594 int max_to_process, const struct timeval *endtime)
1595{
1596 struct event_callback *evcb;
1597 int count = 0;
1598
1599 EVUTIL_ASSERT(activeq != NULL);
1600
1601 for (evcb = TAILQ_FIRST(activeq); evcb; evcb = TAILQ_FIRST(activeq)) {
1602 struct event *ev=NULL;
1603 if (evcb->evcb_flags & EVLIST_INIT) {
1604 ev = event_callback_to_event(evcb);
1605
1606 if (ev->ev_events & EV_PERSIST || ev->ev_flags & EVLIST_FINALIZING)
1607 event_queue_remove_active(base, evcb);
1608 else
1609 event_del_nolock_(ev, EVENT_DEL_NOBLOCK);
1610 event_debug((
1611 "event_process_active: event: %p, %s%s%scall %p",
1612 ev,
1613 ev->ev_res & EV_READ ? "EV_READ " : " ",
1614 ev->ev_res & EV_WRITE ? "EV_WRITE " : " ",
1615 ev->ev_res & EV_CLOSED ? "EV_CLOSED " : " ",
1616 ev->ev_callback));
1617 } else {
1618 event_queue_remove_active(base, evcb);
1619 event_debug(("event_process_active: event_callback %p, "
1620 "closure %d, call %p",
1621 evcb, evcb->evcb_closure, evcb->evcb_cb_union.evcb_callback));
1622 }
1623
1624 if (!(evcb->evcb_flags & EVLIST_INTERNAL))
1625 ++count;
1626
1627
1628 base->current_event = evcb;
1629#ifndef EVENT__DISABLE_THREAD_SUPPORT
1630 base->current_event_waiters = 0;
1631#endif
1632
1633 switch (evcb->evcb_closure) {
1634 case EV_CLOSURE_EVENT_SIGNAL:
1635 EVUTIL_ASSERT(ev != NULL);
1636 event_signal_closure(base, ev);
1637 break;
1638 case EV_CLOSURE_EVENT_PERSIST:
1639 EVUTIL_ASSERT(ev != NULL);
1640 event_persist_closure(base, ev);
1641 break;
1642 case EV_CLOSURE_EVENT: {
1643 void (*evcb_callback)(evutil_socket_t, short, void *);
1644 short res;
1645 EVUTIL_ASSERT(ev != NULL);
1646 evcb_callback = *ev->ev_callback;
1647 res = ev->ev_res;
1648 EVBASE_RELEASE_LOCK(base, th_base_lock);
1649 evcb_callback(ev->ev_fd, res, ev->ev_arg);
1650 }
1651 break;
1652 case EV_CLOSURE_CB_SELF: {
1653 void (*evcb_selfcb)(struct event_callback *, void *) = evcb->evcb_cb_union.evcb_selfcb;
1654 EVBASE_RELEASE_LOCK(base, th_base_lock);
1655 evcb_selfcb(evcb, evcb->evcb_arg);
1656 }
1657 break;
1658 case EV_CLOSURE_EVENT_FINALIZE:
1659 case EV_CLOSURE_EVENT_FINALIZE_FREE: {
1660 void (*evcb_evfinalize)(struct event *, void *);
1661 int evcb_closure = evcb->evcb_closure;
1662 EVUTIL_ASSERT(ev != NULL);
1663 base->current_event = NULL;
1664 evcb_evfinalize = ev->ev_evcallback.evcb_cb_union.evcb_evfinalize;
1665 EVUTIL_ASSERT((evcb->evcb_flags & EVLIST_FINALIZING));
1666 EVBASE_RELEASE_LOCK(base, th_base_lock);
1667 evcb_evfinalize(ev, ev->ev_arg);
1668 event_debug_note_teardown_(ev);
1669 if (evcb_closure == EV_CLOSURE_EVENT_FINALIZE_FREE)
1670 mm_free(ev);
1671 }
1672 break;
1673 case EV_CLOSURE_CB_FINALIZE: {
1674 void (*evcb_cbfinalize)(struct event_callback *, void *) = evcb->evcb_cb_union.evcb_cbfinalize;
1675 base->current_event = NULL;
1676 EVUTIL_ASSERT((evcb->evcb_flags & EVLIST_FINALIZING));
1677 EVBASE_RELEASE_LOCK(base, th_base_lock);
1678 evcb_cbfinalize(evcb, evcb->evcb_arg);
1679 }
1680 break;
1681 default:
1682 EVUTIL_ASSERT(0);
1683 }
1684
1685 EVBASE_ACQUIRE_LOCK(base, th_base_lock);
1686 base->current_event = NULL;
1687#ifndef EVENT__DISABLE_THREAD_SUPPORT
1688 if (base->current_event_waiters) {
1689 base->current_event_waiters = 0;
1690 EVTHREAD_COND_BROADCAST(base->current_event_cond);
1691 }
1692#endif
1693
1694 if (base->event_break)
1695 return -1;
1696 if (count >= max_to_process)
1697 return count;
1698 if (count && endtime) {
1699 struct timeval now;
1700 update_time_cache(base);
1701 gettime(base, &now);
1702 if (evutil_timercmp(&now, endtime, >=))
1703 return count;
1704 }
1705 if (base->event_continue)
1706 break;
1707 }
1708 return count;
1709}

1711/*
1712 * Active events are stored in priority queues. Lower priorities are always
1713 * process before higher priorities. Low priority events can starve high
1714 * priority ones.
1715 */
1716
1717static int
1718event_process_active(struct event_base *base)
1719{
1720 /* Caller must hold th_base_lock */
1721 struct evcallback_list *activeq = NULL;
1722 int i, c = 0;
1723 const struct timeval *endtime;
1724 struct timeval tv;
1725 const int maxcb = base->max_dispatch_callbacks;
1726 const int limit_after_prio = base->limit_callbacks_after_prio;
1727 if (base->max_dispatch_time.tv_sec >= 0) {
1728 update_time_cache(base);
1729 gettime(base, &tv);
1730 evutil_timeradd(&base->max_dispatch_time, &tv, &tv);
1731 endtime = &tv;
1732 } else {
1733 endtime = NULL;
1734 }
1735
1736 for (i = 0; i < base->nactivequeues; ++i) {
1737 if (TAILQ_FIRST(&base->activequeues[i]) != NULL) {
1738 base->event_running_priority = i;
1739 activeq = &base->activequeues[i];
1740 if (i < limit_after_prio)
1741 c = event_process_active_single_queue(base, activeq,
1742 INT_MAX, NULL);
1743 else
1744 c = event_process_active_single_queue(base, activeq,
1745 maxcb, endtime);
1746 if (c < 0) {
1747 goto done;
1748 } else if (c > 0)
1749 break; /* Processed a real event; do not
1750 * consider lower-priority events */
1751 /* If we get here, all of the events we processed
1752 * were internal. Continue. */
1753 }
1754 }
1755
1756done:
1757 base->event_running_priority = -1;
1758
1759 return c;
1760}

1879
1880int
1881event_base_loop(struct event_base *base, int flags)
1882{
1883 const struct eventop *evsel = base->evsel;
1884 struct timeval tv;
1885 struct timeval *tv_p;
1886 int res, done, retval = 0;
1887
1888 /* Grab the lock. We will release it inside evsel.dispatch, and again
1889 * as we invoke user callbacks. */
1890 EVBASE_ACQUIRE_LOCK(base, th_base_lock);
1891
1892 if (base->running_loop) {
1893 event_warnx("%s: reentrant invocation. Only one event_base_loop"
1894 " can run on each event_base at once.", __func__);
1895 EVBASE_RELEASE_LOCK(base, th_base_lock);
1896 return -1;
1897 }
1898
1899 base->running_loop = 1;
1900
1901 clear_time_cache(base);
1902
1903 if (base->sig.ev_signal_added && base->sig.ev_n_signals_added)
1904 evsig_set_base_(base);
1905
1906 done = 0;
1907
1908#ifndef EVENT__DISABLE_THREAD_SUPPORT
1909 base->th_owner_id = EVTHREAD_GET_ID();
1910#endif
1911
1912 base->event_gotterm = base->event_break = 0;
1913
1914 while (!done) {
1915 base->event_continue = 0;
1916 base->n_deferreds_queued = 0;
1917
1918 /* Terminate the loop if we have been asked to */
1919 if (base->event_gotterm) {
1920 break;
1921 }
1922
1923 if (base->event_break) {
1924 break;
1925 }
1926
1927 tv_p = &tv;
1928 if (!N_ACTIVE_CALLBACKS(base) && !(flags & EVLOOP_NONBLOCK)) {
1929 timeout_next(base, &tv_p);
1930 } else {
1931 /*
1932 * if we have active events, we just poll new events
1933 * without waiting.
1934 */
1935 evutil_timerclear(&tv);
1936 }
1937
1938 /* If we have no events, we just exit */
1939 if (0==(flags&EVLOOP_NO_EXIT_ON_EMPTY) &&
1940 !event_haveevents(base) && !N_ACTIVE_CALLBACKS(base)) {
1941 event_debug(("%s: no events registered.", __func__));
1942 retval = 1;
1943 goto done;
1944 }
1945
1946 event_queue_make_later_events_active(base);
1947
1948 clear_time_cache(base);
1949
1950 res = evsel->dispatch(base, tv_p);
1951
1952 if (res == -1) {
1953 event_debug(("%s: dispatch returned unsuccessfully.",
1954 __func__));
1955 retval = -1;
1956 goto done;
1957 }
1958
1959 update_time_cache(base);
1960
1961 timeout_process(base);
1962
1963 if (N_ACTIVE_CALLBACKS(base)) {
1964 int n = event_process_active(base);
1965 if ((flags & EVLOOP_ONCE)
1966 && N_ACTIVE_CALLBACKS(base) == 0
1967 && n != 0)
1968 done = 1;
1969 } else if (flags & EVLOOP_NONBLOCK)
1970 done = 1;
1971 }
1972 event_debug(("%s: asked to terminate loop.", __func__));
1973
1974done:
1975 clear_time_cache(base);
1976 base->running_loop = 0;
1977
1978 EVBASE_RELEASE_LOCK(base, th_base_lock);
1979
1980 return (retval);
1981}

2007/* Schedules an event once */
2008int
2009event_base_once(struct event_base *base, evutil_socket_t fd, short events,
2010 void (*callback)(evutil_socket_t, short, void *),
2011 void *arg, const struct timeval *tv)
2012{
2013 struct event_once *eonce;
2014 int res = 0;
2015 int activate = 0;
2016
2017 /* We cannot support signals that just fire once, or persistent
2018 * events. */
2019 if (events & (EV_SIGNAL|EV_PERSIST))
2020 return (-1);
2021
2022 if ((eonce = mm_calloc(1, sizeof(struct event_once))) == NULL)
2023 return (-1);
2024
2025 eonce->cb = callback;
2026 eonce->arg = arg;
2027
2028 if ((events & (EV_TIMEOUT|EV_SIGNAL|EV_READ|EV_WRITE|EV_CLOSED)) == EV_TIMEOUT) {
2029 evtimer_assign(&eonce->ev, base, event_once_cb, eonce);
2030
2031 if (tv == NULL || ! evutil_timerisset(tv)) {
2032 /* If the event is going to become active immediately,
2033 * don't put it on the timeout queue. This is one
2034 * idiom for scheduling a callback, so let's make
2035 * it fast (and order-preserving). */
2036 activate = 1;
2037 }
2038 } else if (events & (EV_READ|EV_WRITE|EV_CLOSED)) {
2039 events &= EV_READ|EV_WRITE|EV_CLOSED;
2040
2041 event_assign(&eonce->ev, base, fd, events, event_once_cb, eonce);
2042 } else {
2043 /* Bad event combination */
2044 mm_free(eonce);
2045 return (-1);
2046 }
2047
2048 if (res == 0) {
2049 EVBASE_ACQUIRE_LOCK(base, th_base_lock);
2050 if (activate)
2051 event_active_nolock_(&eonce->ev, EV_TIMEOUT, 1);
2052 else
2053 res = event_add_nolock_(&eonce->ev, tv, 0);
2054
2055 if (res != 0) {
2056 mm_free(eonce);
2057 return (res);
2058 } else {
2059 LIST_INSERT_HEAD(&base->once_events, eonce, next_once);
2060 }
2061 EVBASE_RELEASE_LOCK(base, th_base_lock);
2062 }
2063
2064 return (0);
2065}

2067int
2068event_assign(struct event *ev, struct event_base *base, evutil_socket_t fd, short events, void (*callback)(evutil_socket_t, short, void *), void *arg)
2069{

2545/* Implementation function to add an event. Works just like event_add,
2546 * except: 1) it requires that we have the lock. 2) if tv_is_absolute is set,
2547 * we treat tv as an absolute time, not as an interval to add to the current
2548 * time */
2549int
2550event_add_nolock_(struct event *ev, const struct timeval *tv,
2551 int tv_is_absolute)
2552{
2553 struct event_base *base = ev->ev_base;
2554 int res = 0;
2555 int notify = 0;
2556
2557 EVENT_BASE_ASSERT_LOCKED(base);
2558 event_debug_assert_is_setup_(ev);
2559
2560 event_debug((
2561 "event_add: event: %p (fd "EV_SOCK_FMT"), %s%s%s%scall %p",
2562 ev,
2563 EV_SOCK_ARG(ev->ev_fd),
2564 ev->ev_events & EV_READ ? "EV_READ " : " ",
2565 ev->ev_events & EV_WRITE ? "EV_WRITE " : " ",
2566 ev->ev_events & EV_CLOSED ? "EV_CLOSED " : " ",
2567 tv ? "EV_TIMEOUT " : " ",
2568 ev->ev_callback));
2569
2570 EVUTIL_ASSERT(!(ev->ev_flags & ~EVLIST_ALL));
2571
2572 if (ev->ev_flags & EVLIST_FINALIZING) {
2573 /* XXXX debug */
2574 return (-1);
2575 }
2576
2577 /*
2578 * prepare for timeout insertion further below, if we get a
2579 * failure on any step, we should not change any state.
2580 */
2581 if (tv != NULL && !(ev->ev_flags & EVLIST_TIMEOUT)) {
2582 if (min_heap_reserve_(&base->timeheap,
2583 1 + min_heap_size_(&base->timeheap)) == -1)
2584 return (-1); /* ENOMEM == errno */
2585 }
2586
2587 /* If the main thread is currently executing a signal event's
2588 * callback, and we are not the main thread, then we want to wait
2589 * until the callback is done before we mess with the event, or else
2590 * we can race on ev_ncalls and ev_pncalls below. */
2591#ifndef EVENT__DISABLE_THREAD_SUPPORT
2592 if (base->current_event == event_to_event_callback(ev) &&
2593 (ev->ev_events & EV_SIGNAL)
2594 && !EVBASE_IN_THREAD(base)) {
2595 ++base->current_event_waiters;
2596 EVTHREAD_COND_WAIT(base->current_event_cond, base->th_base_lock);
2597 }
2598#endif
2599
2600 if ((ev->ev_events & (EV_READ|EV_WRITE|EV_CLOSED|EV_SIGNAL)) &&
2601 !(ev->ev_flags & (EVLIST_INSERTED|EVLIST_ACTIVE|EVLIST_ACTIVE_LATER))) {
2602 if (ev->ev_events & (EV_READ|EV_WRITE|EV_CLOSED))
2603 res = evmap_io_add_(base, ev->ev_fd, ev);
2604 else if (ev->ev_events & EV_SIGNAL)
2605 res = evmap_signal_add_(base, (int)ev->ev_fd, ev);
2606 if (res != -1)
2607 event_queue_insert_inserted(base, ev);
2608 if (res == 1) {
2609 /* evmap says we need to notify the main thread. */
2610 notify = 1;
2611 res = 0;
2612 }
2613 }
2614
2615 /*
2616 * we should change the timeout state only if the previous event
2617 * addition succeeded.
2618 */
2619 if (res != -1 && tv != NULL) {
2620 struct timeval now;
2621 int common_timeout;
2622#ifdef USE_REINSERT_TIMEOUT
2623 int was_common;
2624 int old_timeout_idx;
2625#endif
2626
2627 /*
2628 * for persistent timeout events, we remember the
2629 * timeout value and re-add the event.
2630 *
2631 * If tv_is_absolute, this was already set.
2632 */
2633 if (ev->ev_closure == EV_CLOSURE_EVENT_PERSIST && !tv_is_absolute)
2634 ev->ev_io_timeout = *tv;
2635
2636#ifndef USE_REINSERT_TIMEOUT
2637 if (ev->ev_flags & EVLIST_TIMEOUT) {
2638 event_queue_remove_timeout(base, ev);
2639 }
2640#endif
2641
2642 /* Check if it is active due to a timeout. Rescheduling
2643 * this timeout before the callback can be executed
2644 * removes it from the active list. */
2645 if ((ev->ev_flags & EVLIST_ACTIVE) &&
2646 (ev->ev_res & EV_TIMEOUT)) {
2647 if (ev->ev_events & EV_SIGNAL) {
2648 /* See if we are just active executing
2649 * this event in a loop
2650 */
2651 if (ev->ev_ncalls && ev->ev_pncalls) {
2652 /* Abort loop */
2653 *ev->ev_pncalls = 0;
2654 }
2655 }
2656
2657 event_queue_remove_active(base, event_to_event_callback(ev));
2658 }
2659
2660 gettime(base, &now);
2661
2662 common_timeout = is_common_timeout(tv, base);
2663#ifdef USE_REINSERT_TIMEOUT
2664 was_common = is_common_timeout(&ev->ev_timeout, base);
2665 old_timeout_idx = COMMON_TIMEOUT_IDX(&ev->ev_timeout);
2666#endif
2667
2668 if (tv_is_absolute) {
2669 ev->ev_timeout = *tv;
2670 } else if (common_timeout) {
2671 struct timeval tmp = *tv;
2672 tmp.tv_usec &= MICROSECONDS_MASK;
2673 evutil_timeradd(&now, &tmp, &ev->ev_timeout);
2674 ev->ev_timeout.tv_usec |=
2675 (tv->tv_usec & ~MICROSECONDS_MASK);
2676 } else {
2677 evutil_timeradd(&now, tv, &ev->ev_timeout);
2678 }
2679
2680 event_debug((
2681 "event_add: event %p, timeout in %d seconds %d useconds, call %p",
2682 ev, (int)tv->tv_sec, (int)tv->tv_usec, ev->ev_callback));
2683
2684#ifdef USE_REINSERT_TIMEOUT
2685 event_queue_reinsert_timeout(base, ev, was_common, common_timeout, old_timeout_idx);
2686#else
2687 event_queue_insert_timeout(base, ev);
2688#endif
2689
2690 if (common_timeout) {
2691 struct common_timeout_list *ctl =
2692 get_common_timeout_list(base, &ev->ev_timeout);
2693 if (ev == TAILQ_FIRST(&ctl->events)) {
2694 common_timeout_schedule(ctl, &now, ev);
2695 }
2696 } else {
2697 struct event* top = NULL;
2698 /* See if the earliest timeout is now earlier than it
2699 * was before: if so, we will need to tell the main
2700 * thread to wake up earlier than it would otherwise.
2701 * We double check the timeout of the top element to
2702 * handle time distortions due to system suspension.
2703 */
2704 if (min_heap_elt_is_top_(ev))
2705 notify = 1;
2706 else if ((top = min_heap_top_(&base->timeheap)) != NULL &&
2707 evutil_timercmp(&top->ev_timeout, &now, <))
2708 notify = 1;
2709 }
2710 }
2711
2712 /* if we are not in the right thread, we need to wake up the loop */
2713 if (res != -1 && notify && EVBASE_NEED_NOTIFY(base))
2714 evthread_notify_base(base);
2715
2716 event_debug_note_add_(ev);
2717
2718 return (res);
2719}

2760 * "blocking" must be one of the EVENT_DEL_{BLOCK, NOBLOCK, AUTOBLOCK,
2761 * EVEN_IF_FINALIZING} values. See those for more information.
2762 */
2763int
2764event_del_nolock_(struct event *ev, int blocking)
2765{
2766 struct event_base *base;
2767 int res = 0, notify = 0;
2768
2769 event_debug(("event_del: %p (fd "EV_SOCK_FMT"), callback %p",
2770 ev, EV_SOCK_ARG(ev->ev_fd), ev->ev_callback));
2771
2772 /* An event without a base has not been added */
2773 if (ev->ev_base == NULL)
2774 return (-1);
2775
2776 EVENT_BASE_ASSERT_LOCKED(ev->ev_base);
2777
2778 if (blocking != EVENT_DEL_EVEN_IF_FINALIZING) {
2779 if (ev->ev_flags & EVLIST_FINALIZING) {
2780 /* XXXX Debug */
2781 return 0;
2782 }
2783 }
2784
2785 base = ev->ev_base;
2786
2787 EVUTIL_ASSERT(!(ev->ev_flags & ~EVLIST_ALL));
2788
2789 /* See if we are just active executing this event in a loop */
2790 if (ev->ev_events & EV_SIGNAL) {
2791 if (ev->ev_ncalls && ev->ev_pncalls) {
2792 /* Abort loop */
2793 *ev->ev_pncalls = 0;
2794 }
2795 }
2796
2797 if (ev->ev_flags & EVLIST_TIMEOUT) {
2798 /* NOTE: We never need to notify the main thread because of a
2799 * deleted timeout event: all that could happen if we don't is
2800 * that the dispatch loop might wake up too early. But the
2801 * point of notifying the main thread _is_ to wake up the
2802 * dispatch loop early anyway, so we wouldn't gain anything by
2803 * doing it.
2804 */
2805 event_queue_remove_timeout(base, ev);
2806 }
2807
2808 if (ev->ev_flags & EVLIST_ACTIVE)
2809 event_queue_remove_active(base, event_to_event_callback(ev));
2810 else if (ev->ev_flags & EVLIST_ACTIVE_LATER)
2811 event_queue_remove_active_later(base, event_to_event_callback(ev));
2812
2813 if (ev->ev_flags & EVLIST_INSERTED) {
2814 event_queue_remove_inserted(base, ev);
2815 if (ev->ev_events & (EV_READ|EV_WRITE|EV_CLOSED))
2816 res = evmap_io_del_(base, ev->ev_fd, ev);
2817 else
2818 res = evmap_signal_del_(base, (int)ev->ev_fd, ev);
2819 if (res == 1) {
2820 /* evmap says we need to notify the main thread. */
2821 notify = 1;
2822 res = 0;
2823 }
2824 }
2825
2826 /* if we are not in the right thread, we need to wake up the loop */
2827 if (res != -1 && notify && EVBASE_NEED_NOTIFY(base))
2828 evthread_notify_base(base);
2829
2830 event_debug_note_del_(ev);
2831
2832 /* If the main thread is currently executing this event's callback,
2833 * and we are not the main thread, then we want to wait until the
2834 * callback is done before returning. That way, when this function
2835 * returns, it will be safe to free the user-supplied argument.
2836 */
2837#ifndef EVENT__DISABLE_THREAD_SUPPORT
2838 if (blocking != EVENT_DEL_NOBLOCK &&
2839 base->current_event == event_to_event_callback(ev) &&
2840 !EVBASE_IN_THREAD(base) &&
2841 (blocking == EVENT_DEL_BLOCK || !(ev->ev_events & EV_FINALIZE))) {
2842 ++base->current_event_waiters;
2843 EVTHREAD_COND_WAIT(base->current_event_cond, base->th_base_lock);
2844 }
2845#endif
2846
2847 return (res);
2848}

3629int
3630event_base_foreach_event_nolock_(struct event_base *base,
3631 event_base_foreach_event_cb fn, void *arg)
3632{
3633 int r, i;
3634 unsigned u;
3635 struct event *ev;
3636
3637 /* Start out with all the EVLIST_INSERTED events. */
3638 if ((r = evmap_foreach_event_(base, fn, arg)))
3639 return r;
3640
3641 /* Okay, now we deal with those events that have timeouts and are in
3642 * the min-heap. */
3643 for (u = 0; u < base->timeheap.n; ++u) {
3644 ev = base->timeheap.p[u];
3645 if (ev->ev_flags & EVLIST_INSERTED) {
3646 /* we already processed this one */
3647 continue;
3648 }
3649 if ((r = fn(base, ev, arg)))
3650 return r;
3651 }
3652
3653 /* Now for the events in one of the timeout queues.
3654 * the min-heap. */
3655 for (i = 0; i < base->n_common_timeouts; ++i) {
3656 struct common_timeout_list *ctl =
3657 base->common_timeout_queues[i];
3658 TAILQ_FOREACH(ev, &ctl->events,
3659 ev_timeout_pos.ev_next_with_common_timeout) {
3660 if (ev->ev_flags & EVLIST_INSERTED) {
3661 /* we already processed this one */
3662 continue;
3663 }
3664 if ((r = fn(base, ev, arg)))
3665 return r;
3666 }
3667 }
3668
3669 /* Finally, we deal wit all the active events that we haven't touched
3670 * yet. */
3671 for (i = 0; i < base->nactivequeues; ++i) {
3672 struct event_callback *evcb;
3673 TAILQ_FOREACH(evcb, &base->activequeues[i], evcb_active_next) {
3674 if ((evcb->evcb_flags & (EVLIST_INIT|EVLIST_INSERTED|EVLIST_TIMEOUT)) != EVLIST_INIT) {
3675 /* This isn't an event (evlist_init clear), or
3676 * we already processed it. (inserted or
3677 * timeout set */
3678 continue;
3679 }
3680 ev = event_callback_to_event(evcb);
3681 if ((r = fn(base, ev, arg)))
3682 return r;
3683 }
3684 }
3685
3686 return 0;
