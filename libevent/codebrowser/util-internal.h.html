77/* A no-op that tricks the compiler into thinking a condition is used while
78 * definitely not making any code for it. Used to compile out asserts while
79 * avoiding "unused variable" warnings. The "!" forces the compiler to
80 * do the sizeof() on an int, in case "condition" is a bitfield value.

82#define EVUTIL_NIL_CONDITION_(condition) do { \
83 (void)sizeof(!(condition)); \
84} while(0)

86/* Internal use only: macros to match patterns of error codes in a
87 cross-platform way. We need these macros because of two historical
88 reasons: first, nonblocking IO functions are generally written to give an
89 error on the "blocked now, try later" case, so sometimes an error from a
90 read, write, connect, or accept means "no error; just wait for more
91 data," and we need to look at the error code. Second, Windows defines
92 a different set of error codes for sockets. */
93
94#ifndef _WIN32
95
96#if EAGAIN == EWOULDBLOCK
97#define EVUTIL_ERR_IS_EAGAIN(e) \
98 ((e) == EAGAIN)
99#else
100#define EVUTIL_ERR_IS_EAGAIN(e) \
101 ((e) == EAGAIN || (e) == EWOULDBLOCK)
102#endif
103
104/* True iff e is an error that means a read/write operation can be retried. */
105#define EVUTIL_ERR_RW_RETRIABLE(e) \
106 ((e) == EINTR || EVUTIL_ERR_IS_EAGAIN(e))
107/* True iff e is an error that means an connect can be retried. */
108#define EVUTIL_ERR_CONNECT_RETRIABLE(e) \
109 ((e) == EINTR || (e) == EINPROGRESS)
110/* True iff e is an error that means a accept can be retried. */
111#define EVUTIL_ERR_ACCEPT_RETRIABLE(e) \
112 ((e) == EINTR || EVUTIL_ERR_IS_EAGAIN(e) || (e) == ECONNABORTED)
113
114/* True iff e is an error that means the connection was refused */
115#define EVUTIL_ERR_CONNECT_REFUSED(e) \
116 ((e) == ECONNREFUSED)
117
118#else
119/* Win32 */
120
121#define EVUTIL_ERR_IS_EAGAIN(e) \
122 ((e) == WSAEWOULDBLOCK || (e) == EAGAIN)
123
124#define EVUTIL_ERR_RW_RETRIABLE(e) \
125 ((e) == WSAEWOULDBLOCK || \
126 (e) == WSAEINTR)
127
128#define EVUTIL_ERR_CONNECT_RETRIABLE(e) \
129 ((e) == WSAEWOULDBLOCK || \
130 (e) == WSAEINTR || \
131 (e) == WSAEINPROGRESS || \
132 (e) == WSAEINVAL)
133
134#define EVUTIL_ERR_ACCEPT_RETRIABLE(e) \
135 EVUTIL_ERR_RW_RETRIABLE(e)
136
137#define EVUTIL_ERR_CONNECT_REFUSED(e) \
138 ((e) == WSAECONNREFUSED)

159/* Helper: Verify that all the elements in 'dlist' are internally consistent.
160 * Checks for circular lists and bad prev/next pointers.
161 *
162 * Example usage:
163 * EVUTIL_ASSERT_LIST_OK(eventlist, event, ev_next);
164 */
165#define EVUTIL_ASSERT_LIST_OK(dlist, type, field) do { \
166 struct type *elm1, *elm2, **nextp; \
167 if (LIST_EMPTY((dlist))) \
168 break; \
169 \
170 /* Check list for circularity using Floyd's */ \
171 /* 'Tortoise and Hare' algorithm */ \
172 elm1 = LIST_FIRST((dlist)); \
173 elm2 = LIST_NEXT(elm1, field); \
174 while (elm1 && elm2) { \
175 EVUTIL_ASSERT(elm1 != elm2); \
176 elm1 = LIST_NEXT(elm1, field); \
177 elm2 = LIST_NEXT(elm2, field); \
178 if (!elm2) \
179 break; \
180 EVUTIL_ASSERT(elm1 != elm2); \
181 elm2 = LIST_NEXT(elm2, field); \
182 } \
183 \
184 /* Now check next and prev pointers for consistency. */ \
185 nextp = &LIST_FIRST((dlist)); \
186 elm1 = LIST_FIRST((dlist)); \
187 while (elm1) { \
188 EVUTIL_ASSERT(*nextp == elm1); \
189 EVUTIL_ASSERT(nextp == elm1->field.le_prev); \
190 nextp = &LIST_NEXT(elm1, field); \
191 elm1 = *nextp; \
192 } \
193 } while (0)
194
195/* Helper: Verify that all the elements in a TAILQ are internally consistent.
196 * Checks for circular lists and bad prev/next pointers.
197 *
198 * Example usage:
199 * EVUTIL_ASSERT_TAILQ_OK(activelist, event, ev_active_next);
200 */
201#define EVUTIL_ASSERT_TAILQ_OK(tailq, type, field) do { \
202 struct type *elm1, *elm2, **nextp; \
203 if (TAILQ_EMPTY((tailq))) \
204 break; \
205 \
206 /* Check list for circularity using Floyd's */ \
207 /* 'Tortoise and Hare' algorithm */ \
208 elm1 = TAILQ_FIRST((tailq)); \
209 elm2 = TAILQ_NEXT(elm1, field); \
210 while (elm1 && elm2) { \
211 EVUTIL_ASSERT(elm1 != elm2); \
212 elm1 = TAILQ_NEXT(elm1, field); \
213 elm2 = TAILQ_NEXT(elm2, field); \
214 if (!elm2) \
215 break; \
216 EVUTIL_ASSERT(elm1 != elm2); \
217 elm2 = TAILQ_NEXT(elm2, field); \
218 } \
219 \
220 /* Now check next and prev pointers for consistency. */ \
221 nextp = &TAILQ_FIRST((tailq)); \
222 elm1 = TAILQ_FIRST((tailq)); \
223 while (elm1) { \
224 EVUTIL_ASSERT(*nextp == elm1); \
225 EVUTIL_ASSERT(nextp == elm1->field.tqe_prev); \
226 nextp = &TAILQ_NEXT(elm1, field); \
227 elm1 = *nextp; \
228 } \
229 EVUTIL_ASSERT(nextp == (tailq)->tqh_last); \
230 } while (0)

259/** Helper macro. If we know that a given pointer points to a field in a
260 structure, return a pointer to the structure itself. Used to implement
261 our half-baked C OO. Example:
262
263 struct subtype {
264 int x;
265 struct supertype common;
266 int y;
267 };
268 ...
269 void fn(struct supertype *super) {
270 struct subtype *sub = EVUTIL_UPCAST(super, struct subtype, common);
271 ...
272 }
273 */
274#define EVUTIL_UPCAST(ptr, type, field) \
275 ((type *)(((char*)(ptr)) - evutil_offsetof(type, field)))

307/* Initialize the state of a week random number generator based on 'seed'. If
308 * the seed is 0, construct a new seed based on not-very-strong platform
309 * entropy, like the PID and the time of day.
310 *
311 * This function, and the other evutil_weakrand* functions, are meant for
312 * speed, not security or statistical strength. If you need a RNG which an
313 * attacker can't predict, or which passes strong statistical tests, use the
314 * evutil_secure_rng* functions instead.
315 */

344/* Replacement for assert() that calls event_errx on failure. */
345#ifdef NDEBUG
346#define EVUTIL_ASSERT(cond) EVUTIL_NIL_CONDITION_(cond)
347#define EVUTIL_FAILURE_CHECK(cond) 0
348#else
349#define EVUTIL_ASSERT(cond) \
350 do { \
351 if (EVUTIL_UNLIKELY(!(cond))) { \
352 event_errx(EVENT_ERR_ABORT_, \
353 "%s:%d: Assertion %s failed in %s", \
354 __FILE__,__LINE__,#cond,__func__); \
355 /* In case a user-supplied handler tries to */ \
356 /* return control to us, log and abort here. */ \
357 (void)fprintf(stderr, \
358 "%s:%d: Assertion %s failed in %s", \
359 __FILE__,__LINE__,#cond,__func__); \
360 abort(); \
361 } \
362 } while (0)
363#define EVUTIL_FAILURE_CHECK(cond) EVUTIL_UNLIKELY(cond)
364#endif
